        function computeFixedPlanBaseCost(planDef) {
            const adjRate = getRate(planDef.rateType);
            const adjCost = (planDef.adjustments || 0) * adjRate;
            const examCost = (planDef.progressExams || 0) * PRICES.progressExam.withPlan;
            const xrayCost = (planDef.xrays || 0) * PRICES.xray;
            return adjCost + examCost + xrayCost;
        }
        function computeOngoingMonthly(planDef) {
            const rate = getRate(planDef.rateType);
            if (planDef.type === "weekly") return Math.round((52 * rate) / 12);
            if (planDef.type === "biweekly") return Math.round((26 * rate) / 12);
            if (planDef.type === "monthly") return Math.round(rate);
            if (planDef.type === "initialWeekly") {
                const annual = (52 * rate) + (3 * PRICES.progressExam.withPlan) + PRICES.xray;
                return Math.round(annual / 12);
            }
            return null;
        }

        function calculateTotal() {
            const planKey = getSelectedPlanKey();
            const planDef = PLANS[planKey];
            const totalCostDiv = document.getElementById('totalCost');
            const additionalCostDiv = document.getElementById('additionalCost');
            const monthlyPaymentsDiv = document.getElementById('monthlyPayments');
            const financialInfo = document.querySelector('.financial-info');

            totalCostDiv.style.display = 'block';
            additionalCostDiv.style.display = 'block';
            monthlyPaymentsDiv.style.display = 'block';
            totalCostDiv.innerHTML = "";
            additionalCostDiv.innerHTML = "";
            monthlyPaymentsDiv.innerHTML = "";

            const additionalCost = getAdditionalCost();

            if (!planDef) {
                if (additionalCost > 0) {
                    totalCostDiv.innerHTML = `Total: <strong>${money(additionalCost)}</strong>`;
                    monthlyPaymentsDiv.style.display = 'none';
                    additionalCostDiv.style.display = 'none';
                    financialInfo.style.display = 'block';
                } else {
                    totalCostDiv.style.display = 'none';
                    financialInfo.style.display = 'none';
                }
                return;
            }

            if (["weekly", "biweekly", "monthly", "initialWeekly"].includes(planDef.type)) {
                totalCostDiv.style.display = 'none';
                const monthly = computeOngoingMonthly(planDef);
                monthlyPaymentsDiv.innerHTML = `Monthly Payment: <strong>${money(monthly)}</strong>`;
                if (additionalCost > 0) additionalCostDiv.innerHTML = `Additional Recommendations: <strong>${money(additionalCost)}</strong>`;
                else additionalCostDiv.innerHTML = "";
                financialInfo.style.display = 'block';
                return;
            }

            if (planDef.type === "decompression") {
                const cash = PRICES.decompression.cashPrice;
                const cashPlusAddons = Math.round(cash + additionalCost);
                const financeBase = decompressionPlanValue();
                const financePlusAddonsMonthly = Math.round((financeBase + additionalCost) / PRICES.decompression.financeMonths);
                totalCostDiv.innerHTML = `<table style="width: 100%; border-collapse: collapse; text-align: left;"><tr><th style="border-bottom: 2px solid #ddd; padding: 8px;">Payment Option</th><th style="border-bottom: 2px solid #ddd; padding: 8px;">Amount</th></tr><tr><td style="padding: 8px;">One-Time Payment (Cash / Check / Credit Card)</td><td style="padding: 8px;"><strong>${money(cashPlusAddons)}</strong></td></tr><tr><td style="padding: 8px;">Four Monthly Payments (Same as Cash)</td><td style="padding: 8px;"><strong>${money(Math.round(cashPlusAddons / PRICES.decompression.cashPayments))}</strong></td></tr><tr><td style="padding: 8px;">${PRICES.decompression.financeLabel}</td><td style="padding: 8px;"><strong>${money(financePlusAddonsMonthly)}</strong></td></tr></table>`;
                monthlyPaymentsDiv.style.display = 'none';
                additionalCostDiv.style.display = 'none';
                financialInfo.style.display = 'block';
                return;
            }

            if (planDef.type === "tenpack") {
                const oneTime = (planDef.adjustments || 10) * getRate(planDef.rateType);
                totalCostDiv.innerHTML = `Total: <strong>${money(oneTime + additionalCost)}</strong>`;
                monthlyPaymentsDiv.style.display = 'none';
                additionalCostDiv.style.display = 'none';
                financialInfo.style.display = 'block';
                return;
            }

            if (planDef.type === "fixed16" || planDef.type === "fixed26") {
                const base = computeFixedPlanBaseCost(planDef);
                const total = Math.round(base + additionalCost);
                totalCostDiv.innerHTML = `Total Plan Cost: <strong>${money(total)}</strong>`;
                additionalCostDiv.style.display = 'none';
                const payments = (planDef.type === "fixed16") ? 4 : 6;
                monthlyPaymentsDiv.innerHTML = `${payments} Monthly Payments of: <strong>${money(Math.round(total / payments))}</strong>`;
                financialInfo.style.display = 'block';
                return;
            }
            financialInfo.style.display = 'none';
        }

                setPdfMode(true); setNotesVisibilityForPdf();
                const { jsPDF } = window.jspdf;
                const doc = new jsPDF('p', 'mm', 'a4');
                const captureEl = document.getElementById('printRoot');
                const signatureCanvas = document.getElementById('signatureCanvas');

                html2canvas(captureEl, { scale: 1.5, logging: false, useCORS: true, backgroundColor: '#ffffff' })
                    .then(canvasContent => {
                        const imgDataContent = canvasContent.toDataURL('image/png');
                        const margin = 10;
                        const imgWidth = 190;
                        let imgHeight = (canvasContent.height * imgWidth) / canvasContent.width;
                        const maxContentHeight = 220;
                        if (imgHeight > maxContentHeight) imgHeight = maxContentHeight;
                        doc.addImage(imgDataContent, 'PNG', margin, margin, imgWidth, imgHeight, '', 'FAST');

                        const signatureStartY = imgHeight + margin + 15;
                        const imgDataSignature = signatureCanvas.toDataURL('image/png');
                        doc.setFontSize(12);
                        doc.text("Patient's Signature:", margin + 5, signatureStartY - 5);
                        doc.setLineWidth(0.5);
                        doc.rect(margin, signatureStartY, imgWidth, 40);
                        doc.addImage(imgDataSignature, 'PNG', margin + 5, signatureStartY + 5, imgWidth - 10, 30);
                        doc.save('treatment-plan.pdf');
                        const pdfBlob = doc.output('blob');
                        resolve(pdfBlob);
                    })
                    .catch(err => { console.error('Error generating PDF:', err); alert('An error occurred while generating the PDF.'); reject(err); })
                    .finally(() => { setPdfMode(false); const notesSection = document.getElementById('optional-notes'); if (notesSection) notesSection.classList.remove('hideForPdf'); });
            });

        async function sendToGHL_Restored(pdfBlob) {
            const targetURL = getTargetWebhookURL();
            if (!targetURL) { alert("⚠️ Please select a valid Treatment Location to submit."); throw new Error("No webhook URL found for selected location."); }

            const payload = collectAllFormData();
            const params = new URLSearchParams();
            for (const key in payload) {
                const val = payload[key];
                if (Array.isArray(val)) { params.append(key, val.join(',')); val.forEach(v => params.append(key, v)); }
                else { params.append(key, val); }
            }

            try {
                await fetch(targetURL, { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, mode: 'no-cors', body: params });
                console.log(`✅ GHL Webhook sent successfully to: ${targetURL}`);
            } catch (err) {
                console.error('❌ GHL Webhook Restore Error:', err); throw err;
            }
        }

        function handlePreFilledData() {
            const urlParams = new URLSearchParams(window.location.search);
            const fields = { 'name': 'patientName', 'email': 'patientEmail', 'phone': 'patientPhone', 'location': 'locationSelect', 'complaint': 'chiefComplaint', 'degen': 'spinalDegeneration', 'bilateral': 'bilateralScales' };
            for (const [paramKey, elementId] of Object.entries(fields)) {
                if (urlParams.has(paramKey)) {
                    const rawValue = urlParams.get(paramKey);
                    const element = document.getElementById(elementId);
                    if (element) { let cleanValue = decodeURIComponent(rawValue); if (!cleanValue.includes('{{')) element.value = cleanValue; }
                }
            }
            const idsCleanCheck = ['patientName', 'patientEmail', 'patientPhone', 'chiefComplaint', 'bilateralScales', 'spinalDegeneration'];
            idsCleanCheck.forEach(id => { const el = document.getElementById(id); if (el && (el.value.includes('{{') || el.value.includes('}}'))) el.value = ""; });

            // Handle Orthotics from URL (e.g. &orthotic=cervical,lumbar)
            const orthoticParam = urlParams.get('orthotic');
            if (orthoticParam && !orthoticParam.includes('{{')) {
                const values = decodeURIComponent(orthoticParam).split(',').map(s => s.trim().toLowerCase());
                document.querySelectorAll('input[name="spinalOrthotic"]').forEach(cb => {
                    if (values.includes(cb.value.toLowerCase())) cb.checked = true;
                });
                toggleOrthoticSelects();
            }
        }
    </script>
