// Helpers

function money(n){ return "$" + Math.round(Number(n) || 0); }

function getRate(rateType){
  if (rateType === "family") return PRICES.adjustment.family;
  if (rateType === "plan") return PRICES.adjustment.plan;
  return PRICES.adjustment.payGo;
}

function computeFixedPlanBaseCost(planDef){
  const adjRate = getRate(planDef.rateType);
  const adjCost = (planDef.adjustments || 0) * adjRate;
  const examCost = (planDef.progressExams || 0) * PRICES.progressExam.withPlan;
  const xrayCost = (planDef.xrays || 0) * PRICES.xray;
  return adjCost + examCost + xrayCost;
}

function computeOngoingMonthly(planDef){
  const rate = getRate(planDef.rateType);
  if (planDef.type === "weekly") return Math.round((52 * rate) / 12);
  if (planDef.type === "biweekly") return Math.round((26 * rate) / 12);
  if (planDef.type === "monthly") return Math.round(rate);
  if (planDef.type === "initialWeekly") {
    const annual = (52 * rate) + (3 * PRICES.progressExam.withPlan) + PRICES.xray;
    return Math.round(annual / 12);
  }
  return null;
}

function getAdditionalCost(){
  const includeAdditional = document.getElementById('includeAdditional').checked;
  if (!includeAdditional) return 0;

  let additionalCost = 0;
  document.querySelectorAll('.dropdown').forEach(dropdown => {
    additionalCost += parseFloat(dropdown.value) || 0;
  });
  return additionalCost;
}

// Additional Recommendations dropdowns (3 identical)
// ============================================================
function populateDropdowns() {
  const dropdownContainer = document.getElementById("dropdownContainer");
  dropdownContainer.innerHTML = "";

  for (let i = 0; i < 3; i++) {
    const select = document.createElement("select");
    select.className = "dropdown";
    select.onchange = updateDetails;

    const noneOpt = document.createElement("option");
    noneOpt.value = "0";
    noneOpt.textContent = "None";
    select.appendChild(noneOpt);

    PRICES.addOns.forEach(opt => {
      const o = document.createElement("option");
      o.value = String(opt.price);
      o.textContent = `${opt.label} - $${opt.price}`;
      select.appendChild(o);
    });

    dropdownContainer.appendChild(select);
  }
}

function renderFixedPlanDetails(planDef){
  const adjRate = getRate(planDef.rateType);
  const adjTotal = (planDef.adjustments || 0) * adjRate;
  const examTotal = (planDef.progressExams || 0) * PRICES.progressExam.withPlan;
  const xrayTotal = (planDef.xrays || 0) * PRICES.xray;

  const title = planDef.type === "fixed26"
    ? "<strong>Treatment Plan Details - 26 Week Plan</strong>"
    : "<strong>Treatment Plan Details - 16 Week Plan</strong>";

  const rows = [];
  rows.push([title, ""]);

  const adjLabelBase = planDef.familyDisplay
    ? `${planDef.adjustments} Chiropractic Adjustments (Family Discount)`
    : `${planDef.adjustments} Chiropractic Adjustments`;

  if (planDef.adjustmentLineStyle === "plain") {
    rows.push([`${adjLabelBase} <span class='light-text'>- Cost per adjustment: $${adjRate}</span>`, money(adjTotal)]);
  } else {
    rows.push([`${adjLabelBase} <span class='light-text'>- Cost per adjustment with plan $${adjRate}, normally $${PRICES.adjustment.payGo}</span>`, money(adjTotal)]);
  }

  (planDef.scheduleLines || []).forEach(line => {
    rows.push([`<span class='indent'>${line}</span>`, ""]);
  });

  if ((planDef.progressExams || 0) > 0) {
    const n = planDef.progressExams;
    if (planDef.examLineStyle === "discountedEach") {
      rows.push([`${n} Progress Exams <span class='light-text'>- Discounted at $${PRICES.progressExam.withPlan} each</span>`, money(examTotal)]);
    } else {
      rows.push([`${n} Progress Exams <span class='light-text'>- Only $${PRICES.progressExam.withPlan}, Normally $${PRICES.progressExam.normal}</span>`, money(examTotal)]);
    }
  }

  if ((planDef.xrays || 0) > 0) {
    rows.push(["1 Set Of Comparative X-Rays", money(xrayTotal)]);
  }

  return rows;
}

// Totals / Payments
// ============================================================
function calculateTotal() {
  const planKey = document.getElementById('treatmentOptions').value;
  const planDef = PLANS[planKey];

  const totalCostDiv = document.getElementById('totalCost');
  const additionalCostDiv = document.getElementById('additionalCost');
  const monthlyPaymentsDiv = document.getElementById('monthlyPayments');
  const financialInfo = document.querySelector('.financial-info');

  totalCostDiv.style.display = 'block';
  additionalCostDiv.style.display = 'block';
  monthlyPaymentsDiv.style.display = 'block';
  totalCostDiv.innerHTML = "";
  additionalCostDiv.innerHTML = "";
  monthlyPaymentsDiv.innerHTML = "";

  const additionalCost = getAdditionalCost();

  if (!planDef) {
    if (additionalCost > 0) {
      totalCostDiv.innerHTML = `Total: <strong>${money(additionalCost)}</strong>`;
      monthlyPaymentsDiv.style.display = 'none';
      additionalCostDiv.style.display = 'none';
      financialInfo.style.display = 'block';
    } else {
      totalCostDiv.style.display = 'none';
      additionalCostDiv.style.display = 'none';
      monthlyPaymentsDiv.style.display = 'none';
      financialInfo.style.display = 'none';
    }
    return;
  }

  if (["weekly","biweekly","monthly","initialWeekly"].includes(planDef.type)) {
    totalCostDiv.style.display = 'none';
    const monthly = computeOngoingMonthly(planDef);
    monthlyPaymentsDiv.innerHTML = `Monthly Payment: <strong>${money(monthly)}</strong>`;

    if (additionalCost > 0) {
      additionalCostDiv.innerHTML = `Additional Recommendations: <strong>${money(additionalCost)}</strong>`;
    } else {
      additionalCostDiv.innerHTML = "";
    }

    financialInfo.style.display = 'block';
    return;
  }

  if (planDef.type === "decompression") {
    const cash = PRICES.decompression.cashPrice;
    const cashPlusAddons = Math.round(cash + additionalCost);

    const financeBase = decompressionPlanValue();
    const financePlusAddonsMonthly = Math.round((financeBase + additionalCost) / PRICES.decompression.financeMonths);

    totalCostDiv.innerHTML = `
      <table style="width: 100%; border-collapse: collapse; text-align: left;">
        <tr>
          <th style="border-bottom: 2px solid #ddd; padding: 8px;">Payment Option</th>
          <th style="border-bottom: 2px solid #ddd; padding: 8px;">Amount</th>
        </tr>
        <tr>
          <td style="padding: 8px;">One-Time Payment (Cash / Check / Credit Card)</td>
          <td style="padding: 8px;"><strong>${money(cashPlusAddons)}</strong></td>
        </tr>
        <tr>
          <td style="padding: 8px;">Four Monthly Payments (Same as Cash)</td>
          <td style="padding: 8px;"><strong>${money(Math.round(cashPlusAddons / PRICES.decompression.cashPayments))}</strong></td>
        </tr>
        <tr>
          <td style="padding: 8px;">${PRICES.decompression.financeLabel}</td>
          <td style="padding: 8px;"><strong>${money(financePlusAddonsMonthly)}</strong></td>
        </tr>
      </table>
    `;

    monthlyPaymentsDiv.style.display = 'none';
    additionalCostDiv.style.display = 'none';
    financialInfo.style.display = 'block';
    return;
  }

  if (planDef.type === "tenpack") {
    const oneTime = (planDef.adjustments || 10) * getRate(planDef.rateType);
    totalCostDiv.innerHTML = `Total: <strong>${money(oneTime + additionalCost)}</strong>`;
    monthlyPaymentsDiv.style.display = 'none';
    additionalCostDiv.style.display = 'none';
    financialInfo.style.display = 'block';
    return;
  }

  if (planDef.type === "fixed16" || planDef.type === "fixed26") {
    const base = computeFixedPlanBaseCost(planDef);
    const total = Math.round(base + additionalCost);

    totalCostDiv.innerHTML = `Total Plan Cost: <strong>${money(total)}</strong>`;
    additionalCostDiv.style.display = 'none';

    const payments = (planDef.type === "fixed16") ? 4 : 6;
    monthlyPaymentsDiv.innerHTML = `${payments} Monthly Payments of: <strong>${money(Math.round(total / payments))}</strong>`;
    financialInfo.style.display = 'block';
    return;
  }

  financialInfo.style.display = 'none';
}

function getSelectedAddOnsByKey(){
  const include = document.getElementById("includeAdditional").checked;
  if (!include) return [];

  const out = [];
  document.querySelectorAll(".dropdown").forEach(sel => {
    const val = parseFloat(sel.value) || 0;
    if (val <= 0) return;

    const txt = sel.options[sel.selectedIndex].text || "";
    const label = txt.split(" - $")[0].trim();
    if (!label || label.toLowerCase() === "none") return;

    const found = PRICES.addOns.find(x => x.label === label);
    out.push({ key: found ? found.key : null, label });
  });

  const seen = new Set();
  return out.filter(x => {
    const k = (x.label || "").toLowerCase();
    if (seen.has(k)) return false;
    seen.add(k);
    return true;
  });
}

function isDeviceAlreadySelectedAsAddon(){
  const selected = getSelectedAddOnsByKey();
  return selected.some(s => s.key === "spinalOrthotic");
}

// Exclude supplements from Alt Pricing totals
const ALT_EXCLUDED_ADDON_KEYS = new Set(["softTissuePack","orthomega","cardioFlow"]);

function altAddOnLineItems(){
  const selected = getSelectedAddOnsByKey();
  const items = [];
  const unmapped = [];

  selected.forEach(s => {
    const k = s.key;

    if (!k) { unmapped.push(s.label); return; }
    if (ALT_EXCLUDED_ADDON_KEYS.has(k)) return;

    if (k === "correctiveExercise") {
      items.push({ label: "Corrective Exercise Plan (one-time)", qty: 1, unit: ALT_PRICES.correctiveExerciseOneTime });
      return;
    }

    if (k === "footOrthotic") {
      items.push({ label: "Custom Foot Orthotic", qty: 1, unit: 360 }); // same as cash
      return;
    }

    if (k === "spinalOrthotic") {
      items.push({ label: "Recommended Device (one-time)", qty: 1, unit: ALT_PRICES.deviceOneTime.gravityDevice });
      return;
    }

    if (k === "sd5") {
      items.push({ label: "Spinal Decompression Add-on (5 visits)", qty: 5, unit: altDecompressionPerVisitTotal() });
      return;
    }

    if (k === "sd10") {
      items.push({ label: "Spinal Decompression Add-on (10 visits)", qty: 10, unit: altDecompressionPerVisitTotal() });
      return;
    }

    if (k === "pemf10") {
      items.push({ label: "PEMF Therapy (10 sessions)", qty: 10, unit: ALT_PRICES.chiroVisit.pemf });
      return;
    }

    if (k === "handMassage10" || k === "chairMassage10") {
      items.push({ label: `${s.label} (10 sessions @ 15 min)`, qty: 10, unit: ALT_PRICES.chiroVisit.massageOrTraction });
      return;
    }

    if (k === "softwave10") {
      items.push({ label: "SoftWave™ Therapy (10 sessions)", qty: 10, unit: ALT_PRICES.softwave.perVisit });
      return;
    }

    unmapped.push(s.label);
  });

  return { items, unmapped };
}

function altVisitCountsForPlan(planDef){
  if (!planDef) return null;

  if (planDef.type === "fixed16" || planDef.type === "fixed26" || planDef.type === "tenpack") {
    const chiroVisits = planDef.adjustments || 0;
    return { chiroVisits, decompressionVisits: 0, totalVisits: chiroVisits, isMonthlyEstimate: false };
  }

  if (planDef.type === "decompression") {
    const decompressionVisits = PRICES.decompression.sdVisits || 0;
    const chiroVisits = PRICES.decompression.chiroVisits || 0;
    return { chiroVisits, decompressionVisits, totalVisits: chiroVisits + decompressionVisits, isMonthlyEstimate: false };
  }

  // Ongoing plans: monthly estimate only (not annual)
  let visitsPerMonth = null;
  if (planDef.type === "weekly" || planDef.type === "initialWeekly") visitsPerMonth = 52 / 12;
  if (planDef.type === "biweekly") visitsPerMonth = 26 / 12;
  if (planDef.type === "monthly") visitsPerMonth = 1;

  return {
    chiroVisits: visitsPerMonth,
    decompressionVisits: 0,
    totalVisits: visitsPerMonth,
    isMonthlyEstimate: true,
    monthlyLabel: "Monthly Estimated Alt Pricing"
  };
}

function altProgressExamCount(totalVisits){
  const n = ALT_PRICES.progressExam.everyNVisits;
  if (!n || totalVisits <= 0) return 0;
  return Math.floor(totalVisits / n);
}

function shouldIncludeAltXrays(planDef){
  return !!(planDef && (planDef.xrays || 0) > 0);
}

/* Itemized per-service Alt Pricing model */
function buildAltPricingModel(planDef){
  const visits = altVisitCountsForPlan(planDef);
  if (!visits) return null;

  const rows = [];

  rows.push({
    kind: "text",
    html: `<div style="font-size:12px;color:#6b7280;line-height:1.4;margin-bottom:10px;">
      This view illustrates a <strong>per-service</strong> billing style rather than a <strong>per-visit</strong> bundled fee.
      Multiple services may be billed separately during the same appointment, which can significantly increase the total billed amount.
    </div>`
  });

  function addLine(label, qty, unit){
    if (!qty || qty <= 0) return;
    rows.push({ label, qty, unit });
  }

  // One-time device: applies to fixed plans + tenpack + decompression unless already selected as spinal orthotic add-on
  const deviceAutoApplies = ["fixed16","fixed26","tenpack","decompression"].includes(planDef.type);
  if (!visits.isMonthlyEstimate && deviceAutoApplies && !isDeviceAlreadySelectedAsAddon()) {
    rows.push({ label: "Recommended Device (one-time)", qty: 1, unit: ALT_PRICES.deviceOneTime.gravityDevice });
  }

  // Monthly estimate plans
  if (visits.isMonthlyEstimate) {
    const v = visits.chiroVisits;
    const a = ALT_PRICES.chiroVisit;

    addLine("Chiropractic adjustment (5 regions)", v, a.adjustment5Regions);
    addLine("Extremity adjustment", v, a.extremity);
    addLine("Massage therapy / Mechanical traction", v, a.massageOrTraction);
    addLine("PEMF therapy", v, a.pemf);
    addLine("Hot/cold pack", v, a.hotColdPack);
  }

  // Fixed-count plans
  if (!visits.isMonthlyEstimate) {
    const chiroV = visits.chiroVisits || 0;
    const decompV = visits.decompressionVisits || 0;

    // Non-decompression chiropractic visits
    if (chiroV > 0) {
      const a = ALT_PRICES.chiroVisit;
      addLine("Chiropractic adjustment (5 regions)", chiroV, a.adjustment5Regions);
      addLine("Extremity adjustment", chiroV, a.extremity);
      addLine("Massage therapy / Mechanical traction", chiroV, a.massageOrTraction);
      addLine("PEMF therapy", chiroV, a.pemf);
      addLine("Hot/cold pack", chiroV, a.hotColdPack);
    }

    // Decompression visits (NO massage/traction)
    if (decompV > 0) {
      const d = ALT_PRICES.decompressionVisit;
      addLine("Vertebral axial decompression", decompV, d.decompression);
      addLine("Chiropractic adjustment (5 regions)", decompV, d.adjustment5Regions);
      addLine("Extremity adjustment", decompV, d.extremity);
      addLine("PEMF therapy", decompV, d.pemf);
      addLine("Hot/cold pack", decompV, d.hotColdPack);
    }

    // Progress exams every N total visits
    const examCount = altProgressExamCount(visits.totalVisits);
    if (examCount > 0) {
      rows.push({
        label: `Progress exams (every ${ALT_PRICES.progressExam.everyNVisits} visits)`,
        qty: examCount,
        unit: ALT_PRICES.progressExam.price
      });
    }

    // Itemized X-rays if included in plan
    if (shouldIncludeAltXrays(planDef)) {
      rows.push({ label: "Cervical spine X-rays", qty: 1, unit: ALT_PRICES.xrays.cervical });
      rows.push({ label: "Thoracic spine X-rays", qty: 1, unit: ALT_PRICES.xrays.thoracic });
      rows.push({ label: "Lumbar spine X-rays", qty: 1, unit: ALT_PRICES.xrays.lumbar });
    }
  }

  // Add-ons
  const addOn = altAddOnLineItems();
  addOn.items.forEach(it => rows.push({ label: it.label, qty: it.qty, unit: it.unit }));

  // Total
  let total = 0;
  rows.forEach(r => {
    if (!r.label) return;
    total += (Number(r.qty) || 0) * (Number(r.unit) || 0);
  });

  return {
    rows,
    total,
    isMonthly: !!visits.isMonthlyEstimate,
    totalLabel: visits.isMonthlyEstimate ? visits.monthlyLabel : "Total Alt Pricing",
    unmapped: addOn.unmapped
  };
}

function renderAltPricing(){
  const container = document.getElementById("altPricingSection");
  const show = document.getElementById("showAltPricing")?.checked;

  if (!container) return;

  if (!show) {
    container.style.display = "none";
    container.innerHTML = "";
    return;
  }

  const planKey = document.getElementById("treatmentOptions").value;
  const planDef = PLANS[planKey];

  if (!planDef) {
    container.style.display = "block";
    container.innerHTML = `<div style="font-size:12px;color:#6b7280;line-height:1.4;">
      Select a treatment plan to view Alt Pricing.
    </div>`;
    return;
  }

  const built = buildAltPricingModel(planDef);
  if (!built) return;

  let html = `
    <div style="border:1px solid #e5e7eb;border-radius:12px;padding:12px;background:#fff;">
      <div style="font-weight:bold;margin-bottom:8px;">Alt Pricing (Internal)</div>
  `;

  built.rows.forEach(r => { if (r.kind === "text") html += r.html; });

  html += `<table style="width:100%;border-collapse:collapse;font-size:13px;">`;
  built.rows.forEach(r => {
    if (!r.label) return;

    const lineTotal = (Number(r.qty) || 0) * (Number(r.unit) || 0);
    const qtyTxt = built.isMonthly ? (Number(r.qty).toFixed(2)) : r.qty;

    html += `
      <tr>
        <td style="padding:6px 6px;border-bottom:1px solid #f0f2f5;">${r.label}</td>
        <td style="padding:6px 6px;border-bottom:1px solid #f0f2f5;text-align:right;white-space:nowrap;">${qtyTxt} × ${altMoney(r.unit)}</td>
        <td style="padding:6px 6px;border-bottom:1px solid #f0f2f5;text-align:right;white-space:nowrap;font-weight:bold;">${altMoney(lineTotal)}</td>
      </tr>
    `;
  });
  html += `</table>`;

  html += `<div style="margin-top:10px;text-align:right;font-weight:bold;">${built.totalLabel}: ${altMoney(built.total)}</div>`;

  if (built.unmapped && built.unmapped.length) {
    html += `<div style="margin-top:10px;font-size:12px;color:#6b7280;line-height:1.4;">
      Selected add-ons not mapped in Alt Pricing (not included in total):<br>
      ${built.unmapped.map(x => `• ${x}`).join("<br>")}
    </div>`;
  }

  html += `</div>`;
  container.innerHTML = html;
  container.style.display = "block";
}

function cleanFamilyText(s){
  return (s || "").replace(/\(Family Discount\)/gi, "").replace(/\s{2,}/g, " ").trim();
}

function selectedAddOnNames(){
  const includeAdditional = document.getElementById("includeAdditional").checked;
  if (!includeAdditional) return [];
  const names = [];

  document.querySelectorAll(".dropdown").forEach(sel => {
    const val = parseFloat(sel.value) || 0;
    if (val <= 0) return;

    const txt = sel.options[sel.selectedIndex].text || "";
    const nameOnly = txt.split(" - $")[0].trim();
    if (nameOnly && nameOnly.toLowerCase() !== "none") names.push(nameOnly);
  });

  return [...new Set(names)];
}

async function copySummarySoap() {
  const planKey = document.getElementById("treatmentOptions").value;
  const planDef = PLANS[planKey];

  const patient = (document.getElementById("patientName").value || "").trim();
  const provider = (document.getElementById("providerSelect").value || "").trim();
  const date = (document.getElementById("todayDate").value || "").trim();
  const notes = (document.getElementById("notesInput").value || "").trim();

  const lines = [];
  lines.push("TREATMENT PLAN SUMMARY");
  if (patient) lines.push(`Patient: ${patient}`);
  if (provider) lines.push(`Provider: ${provider}`);
  if (date) lines.push(`Date: ${date}`);

  if (planDef) {
    lines.push("");
    lines.push(`Plan: ${cleanFamilyText(planDef.name)}`);
    lines.push("Included:");

    if (planDef.type === "fixed16" || planDef.type === "fixed26") {
      lines.push(`- ${planDef.adjustments} Chiropractic Adjustments`);
      (planDef.scheduleLines || []).forEach(s => lines.push(`- ${s}`));
      if ((planDef.progressExams || 0) > 0) lines.push(`- ${planDef.progressExams} Progress Exams`);
      if ((planDef.xrays || 0) > 0) lines.push(`- 1 Set Of Comparative X-Rays`);
    }

    if (planDef.type === "weekly") {
      lines.push(`- Weekly Chiropractic Adjustments`);
      lines.push(`- Educational Workshop`);
      lines.push(`- Special Consults`);
    }

    if (planDef.type === "biweekly") {
      lines.push(`- Bi-Weekly Chiropractic Adjustments`);
      lines.push(`- Educational Workshop`);
      lines.push(`- Special Consults`);
    }

    if (planDef.type === "monthly") {
      lines.push(`- Monthly Chiropractic Adjustments`);
      lines.push(`- Educational Workshop`);
      lines.push(`- Special Consults`);
    }

    if (planDef.type === "initialWeekly") {
      lines.push(`- Weekly Chiropractic Adjustments`);
      lines.push(`- 3 Progress Exams`);
      lines.push(`- 1 Set Of Comparative X-Rays`);
      lines.push(`- Educational Workshop`);
      lines.push(`- Special Consults`);
    }

    if (planDef.type === "tenpack") {
      lines.push(`- 10 Chiropractic Adjustments`);
      lines.push(`- Educational Workshop`);
      lines.push(`- Special Consults`);
    }

    if (planDef.type === "decompression") {
      lines.push(`- 8 Weeks / ${PRICES.decompression.sdVisits} Spinal Decompression Visits`);
      lines.push(`- 16 Weeks / ${PRICES.decompression.chiroVisits} Chiropractic Visits`);
      lines.push(`- Special Consults and Educational Workshop`);
    }
  }

  const addOns = selectedAddOnNames();
  if (addOns.length) {
    lines.push("");
    lines.push("Additional Recommendations:");
    addOns.forEach(a => lines.push(`- ${a}`));
  }

  if (notes) {
    lines.push("");
    lines.push("Notes:");
    lines.push(notes);
  }

  const output = lines.join("\n");

  try {
    await navigator.clipboard.writeText(output);
    alert("Copied to clipboard.");
  } catch (e) {
    const ta = document.createElement("textarea");
    ta.value = output;
    document.body.appendChild(ta);
    ta.select();
    document.execCommand("copy");
    document.body.removeChild(ta);
    alert("Copied to clipboard.");
  }
}

let signaturePad;

function resizeCanvas() {
  const canvas = document.getElementById("signatureCanvas");
  if (!canvas) return;
  const ratio = Math.max(window.devicePixelRatio || 1, 1);
  canvas.width = canvas.offsetWidth * ratio;
  canvas.height = canvas.offsetHeight * ratio;
  canvas.getContext("2d").scale(ratio, ratio);
}

/* ===== Notes PDF fix helpers ===== */
function setNotesVisibilityForPdf() {
  const notesSection = document.getElementById("optional-notes");
  const notesInput = document.getElementById("notesInput");
  const notesPrint = document.getElementById("notesPrint");
  if (!notesSection || !notesInput || !notesPrint) return;

  const val = (notesInput.value || "");
  const hasNotes = val.trim().length > 0;

  // Hide whole section if empty
  notesSection.classList.toggle("hideForPdf", !hasNotes);

  // If we are including notes, mirror into the print div
  if (hasNotes) {
    notesPrint.textContent = val;
  } else {
    notesPrint.textContent = "";
  }
}

function setPdfMode(on) {
  document.documentElement.classList.toggle("pdfMode", !!on);
}

function generatePDF() {
  if (typeof window.jspdf === 'undefined') {
    alert("Error: PDF generation library (jsPDF) failed to load.");
    return;
  }

  if (!signaturePad || signaturePad.isEmpty()) {
    alert("Please sign before submitting.");
    return;
  }

  // Prepare PDF view
  setPdfMode(true);
  setNotesVisibilityForPdf();

  const { jsPDF } = window.jspdf;
  const doc = new jsPDF("p", "mm", "a4");
  const captureEl = document.getElementById("printRoot");
  const signatureCanvas = document.getElementById("signatureCanvas");

  html2canvas(captureEl, {
    scale: 1.75,
    logging: false,
    useCORS: true,
    backgroundColor: "#ffffff"
  })
    .then(canvasContent => {
      const imgDataContent = canvasContent.toDataURL("image/png");

      const margin = 10;
      const imgWidth = 190;
      let imgHeight = (canvasContent.height * imgWidth) / canvasContent.width;

      const maxContentHeight = 220;
      if (imgHeight > maxContentHeight) imgHeight = maxContentHeight;

      doc.addImage(imgDataContent, "PNG", margin, margin, imgWidth, imgHeight, "", "FAST");

      const signatureStartY = imgHeight + margin + 15;
      const imgDataSignature = signatureCanvas.toDataURL("image/png");

      doc.setFontSize(12);
      doc.text("Patient's Signature:", margin + 5, signatureStartY - 5);
      doc.setLineWidth(0.5);
      doc.rect(margin, signatureStartY, imgWidth, 40);
      doc.addImage(imgDataSignature, "PNG", margin + 5, signatureStartY + 5, imgWidth - 10, 30);

      doc.save("treatment-plan.pdf");
    })
    .catch(err => {
      console.error("Error generating PDF:", err);
      alert("An error occurred while generating the PDF. Please try again.");
    })
    .finally(() => {
      // Restore normal view
      setPdfMode(false);
      const notesSection = document.getElementById("optional-notes");
      if (notesSection) notesSection.classList.remove("hideForPdf");
    });
}
